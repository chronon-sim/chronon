# Sender Framework Tests

# Release and RelWithDebInfo normally define NDEBUG, which used to turn a
# substantial part of the sender suite into no-ops (and even remove calls such
# as assert(out.send(...))).  Assertions in test translation units must remain
# executable in every build type.
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    add_compile_options(-UNDEBUG)
endif()
